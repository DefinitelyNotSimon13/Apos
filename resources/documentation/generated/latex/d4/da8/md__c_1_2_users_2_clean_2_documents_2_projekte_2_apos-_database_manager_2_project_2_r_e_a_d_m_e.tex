 Apos -\/ Database Manager is a Windows application designed to manage SQLite3 databases. It allows users to open, read, and edit tables within a SQLite3 database, as well as execute custom SQLite3 commands.\hypertarget{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_features}{}\doxysection{\texorpdfstring{Features}{Features}}\label{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_features}

\begin{DoxyItemize}
\item {\bfseries{Database Operations\+:}}
\begin{DoxyItemize}
\item Open and manage SQLite3 databases.
\item Execute custom SQLite3 commands.
\item Read and edit tables within the database.
\end{DoxyItemize}
\item {\bfseries{User Interface\+:}}
\begin{DoxyItemize}
\item Graphical User Interface (GUI) created using the Qt framework.
\item Intuitive design for easy navigation.
\end{DoxyItemize}
\end{DoxyItemize}\hypertarget{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_prerequisites}{}\doxysection{\texorpdfstring{Prerequisites}{Prerequisites}}\label{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_prerequisites}

\item Windows Operating System

\begin{DoxyEnumerate}

\item Run the installer and follow the on-\/screen instructions.
\end{DoxyEnumerate}\hypertarget{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_usage}{}\doxysection{\texorpdfstring{Usage}{Usage}}\label{md__c_1_2_users_2_clean_2_documents_2_projekte_2_apos-_database_manager_2_project_2_r_e_a_d_m_e_usage}

\begin{DoxyEnumerate}
\item {\bfseries{Opening a Database\+:}}
\begin{DoxyItemize}
\item Launch the application.
\item Click on the "{}\+Open Database"{} button.
\item Select the desired SQLite3 database file.
\end{DoxyItemize}
\item {\bfseries{Executing Commands\+:}}
\begin{DoxyItemize}
\item Use the "{}\+Command Box"{} to enter custom SQLite3 commands.
\item Click the "{}\+Execute"{} button to execute the command.
\end{DoxyItemize}
\item {\bfseries{Managing Tables\+:}}
\begin{DoxyItemize}
\item Use the GUI to read and edit tables within the opened database.
\item Buttons like "{}\+Add,"{} "{}\+Update,"{} and "{}\+Select Table"{} facilitate table operations.
\end{DoxyItemize}
